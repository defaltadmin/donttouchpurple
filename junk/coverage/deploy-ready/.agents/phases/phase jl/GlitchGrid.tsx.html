import React, { useEffect, useRef } from "react";

// J2: GlitchGrid — static 5×5 grid that occasionally glitches:
// cells swap colors, flash white, then settle. Subtle, not distracting.

const CELL_COLORS = [
  "#c026d3", "#7c3aed", "#2563eb", "#0891b2",
  "#059669", "#ca8a04", "#dc2626",
];

const GRID_COLS = 5;
const GRID_ROWS = 5;
const GLITCH_INTERVAL_MS = 1800; // how often a glitch fires
const FLASH_DURATION_MS  = 120;
const SETTLE_DURATION_MS = 280;

type CellState = "idle" | "flash" | "settling";

interface Cell {
  color: string;
  state: CellState;
  stateUntil: number;
}

export function GlitchGrid() {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const rafRef    = useRef<number | null>(null);
  const cellsRef  = useRef<Cell[]>([]);
  const nextGlitchRef = useRef<number>(0);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    // Init cells
    cellsRef.current = Array.from({ length: GRID_COLS * GRID_ROWS }, () => ({
      color: CELL_COLORS[Math.floor(Math.random() * CELL_COLORS.length)],
      state: "idle" as CellState,
      stateUntil: 0,
    }));

    const resize = () => {
      canvas.width  = window.innerWidth;
      canvas.height = window.innerHeight;
    };
    resize();
    window.addEventListener("resize", resize);

    const tick = (ts: number) => {
      const W = canvas.width;
      const H = canvas.height;

      // Cell dimensions — center the grid
      const cellW = Math.min(W / 8, 48);
      const cellH = cellW;
      const gap   = 6;
      const totalW = GRID_COLS * cellW + (GRID_COLS - 1) * gap;
      const totalH = GRID_ROWS * cellH + (GRID_ROWS - 1) * gap;
      const ox = (W - totalW) / 2;
      const oy = (H - totalH) / 2;

      // Trigger glitch
      if (ts >= nextGlitchRef.current) {
        nextGlitchRef.current = ts + GLITCH_INTERVAL_MS + Math.random() * 800;

        // Pick 1–3 random cells to glitch
        const count = 1 + Math.floor(Math.random() * 2);
        for (let i = 0; i < count; i++) {
          const idx = Math.floor(Math.random() * cellsRef.current.length);
          const cell = cellsRef.current[idx];
          cell.state     = "flash";
          cell.stateUntil = ts + FLASH_DURATION_MS;
          // Queue a settle
          setTimeout(() => {
            if (!cellsRef.current[idx]) return;
            cellsRef.current[idx].color = CELL_COLORS[Math.floor(Math.random() * CELL_COLORS.length)];
            cellsRef.current[idx].state = "settling";
            cellsRef.current[idx].stateUntil = performance.now() + SETTLE_DURATION_MS;
          }, FLASH_DURATION_MS);
        }
      }

      ctx.clearRect(0, 0, W, H);

      cellsRef.current.forEach((cell, i) => {
        const col = i % GRID_COLS;
        const row = Math.floor(i / GRID_COLS);
        const x = ox + col * (cellW + gap);
        const y = oy + row * (cellH + gap);

        // Transition cell state
        if (cell.state !== "idle" && ts >= cell.stateUntil) {
          cell.state = "idle";
        }

        let alpha = 0.18;
        let fill  = cell.color;
        if (cell.state === "flash") {
          fill  = "#ffffff";
          alpha = 0.55;
        } else if (cell.state === "settling") {
          alpha = 0.28;
        }

        ctx.save();
        ctx.globalAlpha = alpha;
        ctx.fillStyle   = fill;
        // Rounded rect
        const r = 4;
        ctx.beginPath();
        ctx.moveTo(x + r, y);
        ctx.lineTo(x + cellW - r, y);
        ctx.quadraticCurveTo(x + cellW, y, x + cellW, y + r);
        ctx.lineTo(x + cellW, y + cellH - r);
        ctx.quadraticCurveTo(x + cellW, y + cellH, x + cellW - r, y + cellH);
        ctx.lineTo(x + r, y + cellH);
        ctx.quadraticCurveTo(x, y + cellH, x, y + cellH - r);
        ctx.lineTo(x, y + r);
        ctx.quadraticCurveTo(x, y, x + r, y);
        ctx.closePath();
        ctx.fill();
        ctx.restore();
      });

      rafRef.current = requestAnimationFrame(tick);
    };

    nextGlitchRef.current = performance.now() + 500;
    rafRef.current = requestAnimationFrame(tick);
    return () => {
      window.removeEventListener("resize", resize);
      if (rafRef.current) cancelAnimationFrame(rafRef.current);
    };
  }, []);

  return (
    <canvas
      ref={canvasRef}
      style={{
        position: "fixed",
        top: 0, left: 0,
        width: "100vw", height: "100vh",
        pointerEvents: "none",
        zIndex: 0,
        opacity: 0.7,
      }}
    />
  );
}
